253  static Command *exec(const string &cmd)
254  {
255  auto result = new Command();
256 
257  if (cmd == "./not-executable"s)
258  {
259  throw runtime_error("popen failed: \""s + cmd + "\""s);
260  }
261 
262  FILE *pipe = popen(cmd.c_str(), "r");
263  if (!pipe)
264  {
265  throw runtime_error("popen failed: \""s + cmd + "\""s);
266  }
267 
268  int c;
269  while ((c = fgetc(pipe)) != EOF)
270  {
271  if (c == '\n')
272  {
273  result->lines += 1;
274  }
275  result->output += c;
276  }
277  // Don't concise below 2 lines. It must be assigned to a variable for

279  int n = pclose(pipe);
280  result->exit_code = WEXITSTATUS(n);
