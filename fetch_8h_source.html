270  {

273  pid_t pid;
274 
275  if (pipe(out_fd) == -1)
276  {
277  throw runtime_error("pipe failed");
278  }
279  if (pipe(err_fd) == -1)
280  {
281  throw runtime_error("pipe failed");
282  }
283 
284  if ((pid = fork()) < 0)
285  {
286  throw runtime_error("fork faliled");
287  }
288  else if (pid == 0)
289  { // child
290  if (close(out_fd[0]) == -1)
291  {
292  throw runtime_error("close failed");
293  }
294  if (close(err_fd[0]) == -1)
295  {
296  throw runtime_error("close failed");
297  }
298  if (dup2(out_fd[1], fileno(stdout)) == -1)
299  {
300  throw runtime_error("dup2 failed");
301  }
302  if (dup2(err_fd[1], fileno(stderr)) == -1)
303  {
304  throw runtime_error("dup2 failed");
305  }
306 
307  vector<char *>v{};
308  split(v, cmd);
309  auto argv = v.data();

311 
312  // If execvp() returns, an error have occured.
313  // As the process terminates immediately, it does not free the dynamically allocated memory tied to the argv.
314  switch (errno)
315  {
316  case ENOENT:
317  exit(127);
318  case EACCES:
319  exit(126);
320  default:
321  throw runtime_error("execvp failed: " +
322  string(strerror(errno)) + ": " + argv[0]);
323  }
324  }
325 
326  // parent
327  if (close(out_fd[1]) == -1)
328  {
329  throw runtime_error("close failed");
330  }
331  if (close(err_fd[1]) == -1)
332  {
333  throw runtime_error("close failed");
334  }

336  if (out == NULL)
337  {

339  }

341  if (err == NULL)

354  }
355 
356  while ((c = fgetc(err)) != EOF)
357  {

359  }
360

363  throw runtime_error("fclose failed: " + string(strerror(errno)));
364  }
365

367  {
368  throw runtime_error("fclose failed: " + string(strerror(errno)));
369  }
